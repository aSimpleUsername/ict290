109 void TexturedPolygons::CreateDisplayList(const int &XYZ, const int &listNo,
110  const GLdouble &xImgSize, const GLdouble &zImgSize,
111  const GLdouble &xStart, const GLdouble &yStart, const GLdouble &zStart,
112  const GLdouble &xTimes, const GLdouble &zTimes)
113 {
114  glNewList(listNo, GL_COMPILE);
115  glBegin(GL_QUADS);
116  switch (XYZ)
117  {
118  case 0:
119  // create list where image is on X and Y axis
120  CreateXtoYTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes, false);
121  break;
122  case 1:
123  // create list where image is on X and Z axis
124  CreateXtoZTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes);
125  break;
126  case 2:
127  // create list where image is on Y and Z axis
128  CreateYtoZTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes, false);
129  break;
130  case 3:
131  // create list where (flipped) image is on X and Z axis
132  CreateYtoZTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes, true);
133  break;
134  case 4:
135  // create list where (flipped) image is on X and Y axis
136  CreateXtoYTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes, true);
137  break;
138  }
139  glEnd();
140  glEndList();
141 }
